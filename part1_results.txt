import requests

class OktaProvider:

        # Perform authorization code flow to obtain access token
        token_url = f"https://{self.domain}/oauth2/v2.0/token"

            "code": authorization_code,

        access_token = response.json()["access_token"]
        return access_token

        user_info_url = f"https://{self.domain}/oauth2/v2.0/userinfo"
        headers = {"Authorization": f"Bearer {access_token}"}
        response = requests.get(user_info_url, headers=headers)
        user_info = response.json()
        return user_info

    def logout(self, access_token):

        logout_url = f"https://{self.domain}/oauth2/v2.0/logout"
        data = {
            "client_id": self.client_id,
            "client_secret": self.client_secret,

        }
        response = requests.post(logout_url, data=data)
        return response.json()
